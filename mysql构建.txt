CREATE TABLE `users` (
  `id` int NOT NULL AUTO_INCREMENT COMMENT '用户主键ID',
  `username` varchar(50) NOT NULL COMMENT '登录用户名，唯一',
  `password` varchar(255) NOT NULL COMMENT '加密密码',
  `phone` varchar(20) NOT NULL COMMENT '手机号',
  `role` tinyint NOT NULL DEFAULT 1 COMMENT '角色：1租客 2房东',
  `status` tinyint DEFAULT 1 COMMENT '账号状态：0禁用 1正常',
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `avatar_url` varchar(100) DEFAULT NULL COMMENT '头像地址',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_username` (`username`),
  KEY `idx_role` (`role`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='系统用户表';


CREATE TABLE `houses` (
  `id` int NOT NULL AUTO_INCREMENT COMMENT '房源ID',
  `landlord_id` int NOT NULL COMMENT '房东用户ID，关联users.id',
  `description` text DEFAULT NULL COMMENT '房源描述',
  `address` varchar(255) NOT NULL COMMENT '详细地址',
  `rent_price` decimal(10,2) NOT NULL COMMENT '月租金',
  `house_type` varchar(30) NOT NULL COMMENT '户型，如一室一厅',
  `area` decimal(8,2) DEFAULT NULL COMMENT '房屋面积㎡',
  `facilities` json DEFAULT NULL COMMENT '配套设施JSON数组',
  `images` json DEFAULT NULL COMMENT '房源图片地址JSON数组',
  `status` tinyint DEFAULT 0 COMMENT '房源状态：0待出租 1已租 2下架',
  `view_count` int DEFAULT 0 COMMENT '浏览次数',
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  `user_id` int NOT NULL COMMENT '发布人ID（同房东ID冗余）',
  PRIMARY KEY (`id`),
  KEY `idx_landlord` (`landlord_id`),
  KEY `idx_rent_price` (`rent_price`),
  KEY `idx_status` (`status`),
  KEY `idx_user_id` (`user_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='房源信息表';



CREATE TABLE `house_collect` (
  `id` int NOT NULL AUTO_INCREMENT COMMENT '收藏记录ID',
  `user_id` int NOT NULL COMMENT '收藏用户ID，关联users.id',
  `house_id` int NOT NULL COMMENT '收藏房源ID，关联houses.id',
  `collect_status` tinyint NOT NULL DEFAULT 1 COMMENT '收藏状态：1收藏 0取消',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '收藏时间',
  `update_time` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  KEY `idx_user` (`user_id`),
  KEY `idx_house` (`house_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户房源收藏表';


CREATE TABLE `appointments` (
  `id` int NOT NULL AUTO_INCREMENT COMMENT '预约ID',
  `house_id` int NOT NULL COMMENT '房源ID，关联houses.id',
  `tenant_id` int NOT NULL COMMENT '租客ID，关联users.id',
  `landlord_id` int NOT NULL COMMENT '房东ID，关联users.id',
  `appointment_time` datetime NOT NULL COMMENT '预约看房时间',
  `status` tinyint DEFAULT 0 COMMENT '预约状态：0待确认 1已同意 2已拒绝 3已完成',
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '预约创建时间',
  PRIMARY KEY (`id`),
  KEY `idx_house` (`house_id`),
  KEY `idx_tenant` (`tenant_id`),
  KEY `idx_landlord` (`landlord_id`),
  KEY `idx_status` (`status`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='看房预约记录表';



CREATE TABLE `contracts` (
  `id` int NOT NULL AUTO_INCREMENT COMMENT '合同主键ID',
  `house_id` int NOT NULL COMMENT '房源ID，关联houses.id',
  `tenant_id` int NOT NULL COMMENT '租客ID，关联users.id',
  `landlord_id` int NOT NULL COMMENT '房东ID，关联users.id',
  `contract_no` varchar(50) NOT NULL COMMENT '合同编号，唯一',
  `start_date` date NOT NULL COMMENT '租赁起始日期',
  `end_date` date NOT NULL COMMENT '租赁结束日期',
  `monthly_rent` decimal(10,2) NOT NULL COMMENT '每月租金',
  `deposit_amount` decimal(10,2) NOT NULL COMMENT '押金金额',
  `content` text DEFAULT NULL COMMENT '合同详细条款',
  `tenant_signed` tinyint DEFAULT 0 COMMENT '租客是否签约：0未签 1已签',
  `landlord_signed` tinyint DEFAULT 0 COMMENT '房东是否签约：0未签 1已签',
  `status` tinyint DEFAULT 0 COMMENT '合同状态：0待签约 1生效中 2已到期 3已解约',
  `signed_at` datetime DEFAULT NULL COMMENT '签约完成时间',
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '合同创建时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_contract_no` (`contract_no`),
  KEY `idx_house` (`house_id`),
  KEY `idx_tenant` (`tenant_id`),
  KEY `idx_landlord` (`landlord_id`),
  KEY `idx_status` (`status`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='租房合同表';



CREATE TABLE `payments` (
  `id` int NOT NULL AUTO_INCREMENT COMMENT '缴费记录ID',
  `contract_id` int NOT NULL COMMENT '关联合同ID，关联contracts.id',
  `amount` decimal(10,2) NOT NULL COMMENT '本次缴费金额',
  `pay_period_start` date NOT NULL COMMENT '缴费周期起始日',
  `pay_period_end` date NOT NULL COMMENT '缴费周期结束日',
  `pay_status` tinyint DEFAULT 0 COMMENT '缴费状态：0待缴 1已缴 2逾期',
  `paid_at` datetime DEFAULT NULL COMMENT '实际缴费时间',
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '账单生成时间',
  PRIMARY KEY (`id`),
  KEY `idx_contract` (`contract_id`),
  KEY `idx_pay_status` (`pay_status`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='租金缴费账单表';



CREATE TABLE `reviews` (
  `id` int NOT NULL AUTO_INCREMENT COMMENT '评价ID',
  `rating` tinyint NOT NULL COMMENT '评分 1~5星',
  `comment` text DEFAULT NULL COMMENT '评价文字内容',
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '评价创建时间',
  `appointment_id` int NOT NULL COMMENT '关联看房预约ID，关联appointments.id',
  PRIMARY KEY (`id`),
  KEY `idx_appointment` (`appointment_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='房源看房评价表';
